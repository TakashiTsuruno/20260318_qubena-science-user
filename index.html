function renderDE() {

  const groups = getGroups();

  const grade = activeTab;

  const isElem = grade.startsWith('小');

  const modeKey = activeMode === 'D' ? 'D' : activeMode === 'E_h' ? 'Eh' : 'Er';


  const titles = {

    D: '案D　―　各グループを使って理科を1問以上解いたことがある生徒（期間累計）',

    E_h: '案E 標準版　―　学年対応コンテンツ（小○ / 中○）を1問以上解いたことがある生徒（期間累計）',

    E_r: '案E 練習問題　―　教科書の練習問題を1問以上解いたことがある生徒（期間累計）',

  };

  document.getElementById('de-title').textContent = titles[activeMode];

  document.getElementById('de-head').innerHTML =

    `<th>グループ</th><th>${shortGrade(grade)}</th>`;


  const sections = [

    {label:'小学（R06系・現行）', dot:'grp-elem', filter: c => G_TYPE[c] === 'elem'},

    {label:'中学（R07系・現行）', dot:'grp-mid',  filter: c => G_TYPE[c] === 'mid'},

    {label:'中学（R03系・旧世代）',dot:'grp-old',  filter: c => G_TYPE[c] === 'old'},

  ];


  let html = '';

  let grandTotal = 0;


  sections.forEach(sec => {

    const secGroups = groups.filter(sec.filter);

    if (!secGroups.length) return;

    const secForElem = sec.label.startsWith('小');

    const secForMid  = sec.label.startsWith('中');

    if (isElem && secForMid)  return;

    if (!isElem && secForElem) return;


    html += `<tr class="row-section"><td colspan="2"><span class="${sec.dot}"></span>${sec.label}</td></tr>`;

    let secTotal = 0;


    secGroups.forEach(c => {

      const d = DE_RAW.find(r => r.g === grade && r.c === c);

      const val = d?.[modeKey];

      const unprovided = (c === 'skyR06' && activeMode === 'E_r') ||

                         ((c === 'tshR03' || c === 'krkR03') && activeMode === 'E_r');

      const display = unprovided

        ? `<span class="na">未提供</span>`

        : (val == null ? `<span class="na">—</span>` : `<span class="num">${val.toLocaleString()}人</span>`);

      if (val != null && !unprovided) secTotal += val;

      html += `<tr><td>${GL[c]}</td><td>${display}</td></tr>`;

    });


    if (secGroups.length > 1) {

      html += `<tr class="row-total"><td>小計</td><td class="num">${secTotal.toLocaleString()}人</td></tr>`;

    }

    grandTotal += secTotal;

  });


  document.getElementById('de-body').innerHTML = html;


  const bdp = BB[grade]?.Bdp || 0;

  const diff = Math.abs(bdp - grandTotal);

  document.getElementById('de-note').innerHTML = activeMode === 'D'

    ? `参考：案B"（${grade}・開始済み週次平均）${bdp.toLocaleString()}人　／　表示グループ合計との差 ▲${diff.toLocaleString()}人　―　週次平均と累計UUの計算方式の違い・グループ未網羅によるもの`

    : activeMode === 'E_r'

      ? 'skyR06（信教・小学）は練習問題未提供。tshR03・krkR03（中学旧世代）は練習問題の記録なし。両コンテンツを使った生徒は標準版・練習問題の両方にカウントされる（重複あり）。'

      : '両コンテンツを使った生徒は標準版・練習問題の両方にカウントされる（重複あり）。学年対応コンテンツ（例：中1在籍→「中1」コンテンツ）のみ集計。';

}
